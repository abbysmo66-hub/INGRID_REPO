/// Ejercicio 02 (Intermedio-Alto) - Verificador de número primo
/// Un laboratorio de criptografía utiliza números primos en sus algoritmos. Se necesita un programa que,
/// dado un número entero positivo ingresado por el usuario, indique si es primo o no, y en caso de no serlo,
/// muestre el primer divisor exacto encontrado (distinto de 1 y del propio número).
/// La verificación debe realizarse en una función que devuelva 1 si el número es primo y 0 si no lo es. El
/// primer divisor se obtiene mediante otra función que lo entrega por referencia. La presentación del
/// resultado la maneja una función separada.

#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include <time.h>
#include <ctype.h>
#include <string.h>
#include <math.h>

int esPrimo (int numero);
void obtenerDivisor (int numero, int *divisor);
void mostrarResultado (int numero, int primo, int divisor);

int main()
{
    int num, primo, divisor=0;
    printf ("Ingrese un numero entero positivo: ");
    scanf ("%d", &num);

    if (num <= 1)
    {
        printf ("Ingrese un numero mayor a 1");
    return 0;
    }
    /// Llamamos a la funcion
    primo = esPrimo(num);
    if (primo==0)
    {
        obtenerDivisor(num,&divisor);
    }
    mostrarResultado (num,primo,divisor);
return 0;
}
// Devuelve 1 si es primo,0 si no es primo.
int esPrimo (int numero)
{
    if (numero<2) return 0;
    for (int i=2; i*i<=numero;i++)
    {
        if (numero % i==0)
        {
            return 0; // se encontro un divisor, no es primo.
        }
    }
    return 1; //es primo.
}
/// Definimos funciones encontrando el primer divisor y lo entrega por referencia
void obtenerDivisor (int numero, int *divisor)
{
    for (int i = 2; i < numero; i++)
    {
        if (numero % i == 0)
        {
            *divisor = i;
        }
    }
return 0; //detiene la busqueda al hallar el primero
}

/// Para obtener los resultados en pantalla
void mostrarResultado (int numero, int primo, int divisor)
{
    if (primo == 1)
    {
        printf ("El numero %d es primo\n", numero);
    }else{
        printf ("El numero %d no es primo\n", numero);
        printf ("El primer divisor exacto encontrado es: %d\n", divisor);
    }
}
